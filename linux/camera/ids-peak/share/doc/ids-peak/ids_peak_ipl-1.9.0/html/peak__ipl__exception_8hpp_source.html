189  != PEAK_IPL_RETURN_CODE_SUCCESS)
190  {
191  throw Exception("Could not query the last error!");
192  }
193  std::vector<char> lastErrorDescription(lastErrorDescriptionSize);
194  if (PEAK_IPL_C_ABI_PREFIX PEAK_IPL_Library_GetLastError(
195  &lastErrorCode, lastErrorDescription.data(), &lastErrorDescriptionSize)
196  != PEAK_IPL_RETURN_CODE_SUCCESS)
197  {
198  throw Exception("Could not query the last error!");
199  }
200 
201  std::stringstream errorText;

203  << StringFromPEAK_IPL_RETURN_CODE(static_cast<PEAK_IPL_RETURN_CODE_t>(lastErrorCode))
204  << ") | Error-Description: " << lastErrorDescription.data() << "]";
205
